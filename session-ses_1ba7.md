// ── Pre-scan ─────────────────────────────────────────

fn run_pre_scan(dir: &Path) -> ScanStats {
    let start = Instant::now();
    let mut stats = ScanStats {
        total_entries: 0,
        total_files: 0,
        total_dirs: 0,
        total_symlinks: 0,
        total_bytes: 0,
        scan_duration: Duration::ZERO,
    };

    for entry in WalkDir::new(dir).follow_links(false) {
        match entry {
            Ok(e) => {
                stats.total_entries += 1;
                let ft = e.file_type();
                if ft.is_file() {
                    stats.total_files += 1;
                    if let Ok(meta) = e.metadata() {
                        stats.total_bytes += meta.len();
                    }
                } else if ft.is_dir() {
                    stats.total_dirs += 1;
                } else if ft.is_symlink() {
                    stats.total_symlinks += 1;
                }
            }
            Err(err) => {
                eprintln!("[warn] scan error: {}", err);
            }
        }
    }

    stats.scan_duration = start.elapsed();
    stats
}

fn format_bytes(bytes: u64) -> String {
    const UNITS: &[&str] = &["B", "KiB", "MiB", "GiB", "TiB"];
    let mut size = bytes as f64;
    let mut unit_idx = 0;
    while size >= 1024.0 && unit_idx < UNITS.len() - 1 {
        size /= 1024.0;
        unit_idx += 1;
    }
    if unit_idx == 0 {
        format!("{} {}", bytes, UNITS[unit_idx])
    } else {
        format!("{:.2} {}", size, UNITS[unit_idx])
    }
}

async fn hash_file(path: &Path) -> Result<String, io::Error> {
    use tokio::io::AsyncReadExt;
    let file = tokio::fs::File::open(path).await?;
    let mut reader = tokio::io::BufReader::new(file);
    let mut hasher = Sha256::new();
    let mut buf = vec![0u8; 64 * 1024];
    loop {
        let n = reader.read(&mut buf).await?;
        if n == 0 {
            break;
        }
        hasher.update(&buf[..n]);
    }
    Ok(format!("{:x}", hasher.finalize()))
}

fn process_entry_sync(entry: &walkdir::DirEntry, base: &Path, absolute: bool) -> FileEntry {
    let path = if absolute {
        entry.path().to_path_buf()
    } else {
        entry.path().strip_prefix(base).unwrap_or(entry.path()).to_path_buf()
    };

    let ft = entry.file_type();
    let (entry_type, sha256, size, modified, link_target) = if ft.is_dir() {
        let meta = entry.metadata().ok();
        (
            EntryType::Dir,
            None,
            None,
            meta.and_then(|m| {
                m.modified().ok().map(|t| {
                    let d: DateTime<Utc> = t.into();
                    d
                })
            }),
            None,
        )
    } else if ft.is_symlink() {
        let target = entry.path().read_link().ok().map(|p| p.to_string_lossy().to_string());
        (
            EntryType::Symlink,
            None,
            None,
            None,
            target,
        )
    } else {
        let meta = entry.metadata().ok();
        (
            EntryType::File,
            None, // filled later
            meta.map(|m| m.len()),
            meta.and_then(|m| {
                m.modified().ok().map(|t| {
                    let d: DateTime<Utc> = t.into();
                    d
                })
            }),
            None,
        )
    };

    FileEntry { path, entry_type, sha256, size, modified, link_target }
}

fn write_csv(entries: &[FileEntry], output: Option<&PathBuf>) -> Result<(), Box<dyn std::error::Error>> {
    let mut writer: Box<dyn io::Write> = match output {
        Some(path) => Box::new(std::fs::File::create(path)?),
        None => Box::new(io::stdout()),
    };

    let mut csv_writer = csv::Writer::from_writer(writer);
    csv_writer.write_record(&["path", "type", "sha256", "size", "modified", "link_target"])?;

    for entry in entries {
        let sha256 = entry.sha256.as_deref().unwrap_or("");
        let size = entry.size.map(|s| s.to_string()).unwrap_or_default();
        let modified = entry.modified.map(|t| t.to_rfc3339()).unwrap_or_default();
        let link_target = entry.link_target.as_deref().unwrap_or_default();
        let entry_type = match entry.entry_type {
            EntryType::File => "file",
            EntryType::Dir => "dir",
            EntryType::Symlink => "symlink",
        };
        csv_writer.write_record(&[
            entry.path.to_string_lossy().as_ref(),
            entry_type,
            sha256,
            &size,
            &modified,
            link_target,
        ])?;
    }

    csv_writer.flush()?;
    Ok(())
}

        stats.total_dirs,
